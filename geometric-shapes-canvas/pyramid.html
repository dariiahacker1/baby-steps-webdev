rgba fill color editor

//We get access to the HTML elements that have the identifiers 'redRange', 'greenRange', 'blueRange', 'alphaRange' using the document.getElementById() method and store them in the variables redRange, greenRange, blueRange, alphaRange.
var redRange = document.getElementById('redRange');
var greenRange = document.getElementById('greenRange');
var blueRange = document.getElementById('blueRange');
var alphaRange = document.getElementById('alphaRange');

//We add an input event handler to each element of the range (slider) to respond to changes in of the values in these elements.
redRange.addEventListener('input', changeFillColor);
greenRange.addEventListener('input', changeFillColor);
blueRange.addEventListener('input', changeFillColor);
alphaRange.addEventListener('input', changeFillColor);

//We create the changeFillColor() function, which is called every time the user changes the value in one of the elements of the range.
function changeFillColor() {
var red = redRange.value;
var green = greenRange.value;
var blue = blueRange.value;
var alpha = alphaRange.value;

//We create a fillColor string containing a color value in rgba() format with the values that we got from the elements of the range.
var fillColor = 'rgba(' + red + ',' + green + ',' + blue + ',' + alpha + ')';

//Set fillColor as a color filler using the fillStyle property of the context drawing ctx.
ctx.fillStyle = fillColor;

//Clear the drawing screen using the clearRect() method of the ctx drawing context. The parameters 0, 0 indicate the starting position of the clearing, and canvas.width and canvas.height indicate the width and height of the canvas element.
ctx.clearRect(0, 0, canvas.width, canvas.height);
}
