Change the fill color

// Creating an input element of type range and assigning the id, min, max and value attributes to it
input type="range" id="colorRange" min="0" max="255" value="136"
//Getting the input element using the document.getElementById() method and assigning it to the to the colorRange variable:
var colorRange = document.getElementById('colorRange');
//Adding an input event handler to the input element using the addEventListener() method. When the slider value changes, the drawShape(), bukva(), drawOutline(), drawLines() functions are called, dopLines(), and drawCircles():
colorRange.addEventListener('input', function() { drawShape(); bukva(); drawOutline(); drawLines(); dopLines(); drawCircles(); });
//Clearing the entire canvas using the clearRect() method so that when you change the color of the shape fill the previous color does not remain:
ctx.clearRect(0, 0, canvas.width, canvas.height);
//Creating a fillColor variable that contains a color value for filling shapes.
var fillColor = 'rgba(' + colorRange.value + ', 64, 216, 0.53)';
//Setting the fillStyle variable of the ctx canvas context to change the fill color of the shapes on the canvas:Setting the fillStyle variable of the ctx canvas context to change the fill color of the shapes on the canvas:
ctx.fillStyle = fillColor;
